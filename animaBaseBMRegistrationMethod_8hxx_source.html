266 bool
267 BaseBMRegistrationMethod <TInputImageType>
268 ::ComposeAddOnWithTransform(TransformPointer &computedTransform, TransformType *addOn)
269 {
270  if (m_Agregator->GetOutputTransformType() != AgregatorType::SVF)
271  {
272  typename TransformType::ParametersType oldPars = computedTransform->GetParameters();
273 
274  AffineTransformType *tmpTrsf = dynamic_cast<AffineTransformType *>(computedTransform.GetPointer());
275  AffineTransformType *tmpAddOn = dynamic_cast<AffineTransformType *>(addOn);
276  tmpTrsf->Compose(tmpAddOn, true);
277 
278  typename TransformType::ParametersType newPars = computedTransform->GetParameters();
279 
280  // Compute the distance between consecutive solutions, until a certain threshold
281  double err = 0;
282  for (unsigned int i = 0; i < newPars.Size(); ++i)
283  err += std::pow(newPars[i] - oldPars[i], 2.);
284 
285  if (err < m_MinimalTransformError)
286  return false;
287  }
288  else
289  {
290  // Add update to current velocity field (cf. Vercauteren et al, 2008)
291  SVFTransformType *tmpTrsf = dynamic_cast<SVFTransformType *>(computedTransform.GetPointer());
292  SVFTransformType *tmpAddOn = dynamic_cast<SVFTransformType *>(addOn);
293 
294  anima::composeSVF(tmpTrsf,tmpAddOn,this->GetNumberOfThreads());
295  if (m_SVFElasticRegSigma > 0)
296  {
297  typedef typename SVFTransformType::VectorFieldType VelocityFieldType;
298 
299  typedef anima::SmoothingRecursiveYvvGaussianImageFilter <VelocityFieldType, VelocityFieldType> SmoothingFilterType;
300  typename SmoothingFilterType::Pointer smootherPtr = SmoothingFilterType::New();
301 
302  smootherPtr->SetInput(tmpTrsf->GetParametersAsVectorField());
303  smootherPtr->SetSigma(m_SVFElasticRegSigma);
304  smootherPtr->SetNumberOfThreads(this->GetNumberOfThreads());
305 
306  smootherPtr->Update();
307 
308  typename VelocityFieldType::Pointer tmpSmoothed = smootherPtr->GetOutput();
309  tmpSmoothed->DisconnectPipeline();
310 
311  tmpTrsf->SetParametersAsVectorField(tmpSmoothed);
312  }
313  }
314
