729  {

731  effectiveNumberOfParticles[m] = 1.0 / effectiveNumberOfParticles[m];
732  else
733  continue; // Q: shouldn't we be treating this case as an empty cluster that shouldn't even exist? (same as previous Q)

736  if (effectiveNumberOfParticles[m] < m_ResamplingThreshold * fiberComputationData.classSizes[m])

738  weightSpecificClassValues.resize(fiberComputationData.classSizes[m]);

779  // fiberTrash.push_back(fiberParticlesCopy[i]);
780  }
781  }
782  }

793  // Example :

795  stopLoop = true;
796

798 
799  for (unsigned int i = 0;i < fiberComputationData.particleWeights.size();++i)
800  {

802  itkExceptionMacro("Nan weights after update class membership");
803  }
804  }
805

810  for (unsigned int i = 0;i < numberOfClasses;++i)

821  else

823

1199  fiberData.particleWeights = newParticleWeights;
