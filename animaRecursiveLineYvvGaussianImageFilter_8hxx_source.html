54 template <typename TInputImage, typename TOutputImage>
55 void
56 RecursiveLineYvvGaussianImageFilter<TInputImage,TOutputImage>
57 ::SetUp(ScalarRealType spacing)
58 {
59  const ScalarRealType sigmad = m_Sigma/spacing;
60 
61  // Compute q according to 16 in Young et al on Gabor filering
62  ScalarRealType q = 0;
63  if (sigmad >= 3.556)
64  q = 0.9804 * (sigmad - 3.556) + 2.5091;
65  else
66  {
67  if (sigmad < 0.5)
68  std::cerr << "Too low sigma value (< 0.5), computation will not be precise." << std::endl;
69 
70  q = 0.0561 * sigmad * sigmad + 0.5784 * sigmad - 0.2568;
71  }
72 
73  // Compute B and B1 to B3 according to Young et al 2003
74  ScalarRealType m0 = 1.16680;
75  ScalarRealType m1 = 1.10783;
76  ScalarRealType m2 = 1.40586;
77  ScalarRealType scale = (m0 + q) * (m1 * m1 + m2 * m2 + 2 * m1 * q + q * q);
78 
79  m_B1 = q * (2 * m0 * m1 + m1 * m1 + m2 * m2 + (2 * m0 + 4 * m1) * q + 3 * q * q) / scale;
80 
81  m_B2 = - q * q * (m0 + 2 * m1 + 3 * q) / scale;
82 
83  m_B3 = q * q * q / scale;
84 
85  ScalarRealType baseB = (m0 * (m1 * m1 + m2 * m2)) / scale;
86  m_B = baseB * baseB;
87 
88  // M Matrix for initialization on backward pass, from Triggs and Sdika, IEEE TSP
89  m_MMatrix = vnl_matrix <ScalarRealType> (3,3);
90 
91  m_MMatrix(0,0) = - m_B3 * m_B1 + 1 - m_B3 * m_B3 - m_B2;
92  m_MMatrix(0,1) = (m_B3 + m_B1) * (m_B2 + m_B3 * m_B1);
93  m_MMatrix(0,2) = m_B3 * (m_B1 + m_B3 * m_B2);
94 
95  m_MMatrix(1,0) = m_B1 + m_B3 * m_B2;
96  m_MMatrix(1,1) = (1 - m_B2) * (m_B2 + m_B3 * m_B1);
97  m_MMatrix(1,2) = - m_B3 * (m_B3 * m_B1 + m_B3 * m_B3 + m_B2 - 1);
98 
99  m_MMatrix(2,0) = m_B3 * m_B1 + m_B2 + m_B1 * m_B1 - m_B2 * m_B2;
100  m_MMatrix(2,1) = m_B1 * m_B2 + m_B3 * m_B2 * m_B2 - m_B1 * m_B3 * m_B3 - m_B3 * m_B3 * m_B3 - m_B3 * m_B2 + m_B3;
101  m_MMatrix(2,2) = m_B3 * (m_B1 + m_B3 * m_B2);
102 
103  m_MMatrix /= (1 + m_B1 - m_B2 + m_B3) * (1 - m_B1 - m_B2 - m_B3) * (1 + m_B2 + (m_B1 - m_B3) * m_B3);
104 }
