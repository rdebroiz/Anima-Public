167 void
168 ResampleImageFilter<TInputImage,TOutputImage,TInterpolatorPrecisionType>
169 ::ThreadedGenerateData(const OutputImageRegionType& outputRegionForThread,
170  itk::ThreadIdType threadId)
171 {
172  // Get the output pointers
173  OutputImagePointer outputPtr = this->GetOutput();
174 
175  // Get ths input pointers
176  InputImageConstPointer inputPtr=this->GetInput();
177 
178  // Create an iterator that will walk the output region for this thread.
179  typedef itk::ImageRegionIteratorWithIndex<TOutputImage> OutputIterator;
180 
181  OutputIterator outIt(outputPtr, outputRegionForThread);
182 
183  // Define a few indices that will be used to translate from an input pixel
184  // to an output pixel
185  PointType outputPoint; // Coordinates of current output pixel
186  PointType inputPoint; // Coordinates of current input pixel
187 
188  typedef itk::ContinuousIndex<TInterpolatorPrecisionType, ImageDimension> ContinuousIndexType;
189  ContinuousIndexType inputIndex;
190 
191  // Support for progress methods/callbacks
192  itk::ProgressReporter progress(this, threadId, outputRegionForThread.GetNumberOfPixels());
193 
194  typedef typename InterpolatorType::OutputType OutputType;
195 
196  // Min/max values of the output pixel type AND these values
197  // represented as the output type of the interpolator
198  const PixelType minValue = itk::NumericTraits<PixelType >::NonpositiveMin();
199  const PixelType maxValue = itk::NumericTraits<PixelType >::max();
200 
201  const OutputType minOutputValue = static_cast<OutputType>(minValue);
202  const OutputType maxOutputValue = static_cast<OutputType>(maxValue);
203 
204  // Walk the output region
205  outIt.GoToBegin();
206 
207  while ( !outIt.IsAtEnd() )
208  {
209  // Determine the index of the current output pixel
210  outputPtr->TransformIndexToPhysicalPoint( outIt.GetIndex(), outputPoint );
211 
212  // Compute corresponding input pixel position
213  inputPoint = m_Transform->TransformPoint(outputPoint);
214  inputPtr->TransformPhysicalPointToContinuousIndex(inputPoint, inputIndex);
215 
216  // Evaluate input at right position and copy to the output
217  if( m_Interpolator->IsInsideBuffer(inputIndex) )
218  {
219  PixelType pixval;
220  OutputType value = m_Interpolator->EvaluateAtContinuousIndex(inputIndex);
221 
222  if (m_ScaleIntensitiesWithJacobian)
223  {
224  double jacobianValue = 1;
225  if (m_LinearTransform)
226  jacobianValue = this->ComputeLinearJacobianValue();
227  else
228  jacobianValue = this->ComputeLocalJacobianValue(outIt.GetIndex());
229 
230  value *= jacobianValue;
231  }
232 
233  if( value < minOutputValue )
234  {
235  pixval = minValue;
236  }
237  else if( value > maxOutputValue )
238  {
239  pixval = maxValue;
240  }
241  else
242  {
243  pixval = static_cast<PixelType>( value );
244  }
245  outIt.Set( pixval );
246  }
247  else
248  {
249  outIt.Set(m_DefaultPixelValue); // default background value
250  }
251 
252  progress.CompletedPixel();
253  ++outIt;
254  }
255 }

432 template <class TInputImage, class TOutputImage, class TInterpolatorPrecisionType>
433 double
434 ResampleImageFilter<TInputImage,TOutputImage,TInterpolatorPrecisionType>
435 ::ComputeLocalJacobianValue(const InputIndexType &index)
436 {
437  InputIndexType startIndDef, endIndDef;
438  startIndDef = this->GetOutput()->GetLargestPossibleRegion().GetIndex();
439  endIndDef = startIndDef + this->GetOutput()->GetLargestPossibleRegion().GetSize();
440 
441  vnl_matrix_fixed <double,ImageDimension,ImageDimension> jacMatrix;
442 
443  vnl_matrix_fixed <double,ImageDimension,ImageDimension> deltaMatrix;
444  deltaMatrix.fill(0);
445  InputIndexType posBef, posAfter;
446  PointType tmpPosBef, tmpPosAfter;
447 
448  vnl_matrix_fixed <double,ImageDimension,ImageDimension> resDiff;
449  resDiff.fill(0);
450  OutputImagePointer outputPtr = this->GetOutput();
451 
452  PointType tmpPos;
453 
454  for (unsigned int i = 0;i < ImageDimension;++i)
455  {
456  posBef = index;
457  posBef[i]--;
458 
459  if (posBef[i] < startIndDef[i])
460  posBef[i] = startIndDef[i];
461 
462  outputPtr->TransformIndexToPhysicalPoint(posBef,tmpPosBef);
463 
464  posAfter = index;
465  posAfter[i]++;
466 
467  if (posAfter[i] >= endIndDef[i])
468  posAfter[i] = endIndDef[i] - 1;
469 
470  outputPtr->TransformIndexToPhysicalPoint(posAfter,tmpPosAfter);
471 
472  if (posAfter[i] == posBef[i])
473  {
474  deltaMatrix(i,i) = 1;
475  continue;
476  }
477 
478  for (unsigned int j = 0;j < ImageDimension;++j)
479  deltaMatrix(i,j) = tmpPosAfter[j] - tmpPosBef[j];
480 
481  tmpPos = m_Transform->TransformPoint(tmpPosAfter);
482  for (unsigned int j = 0;j < ImageDimension;++j)
483  resDiff(i,j) = tmpPos[j];
484 
485  tmpPos = m_Transform->TransformPoint(tmpPosBef);
486  for (unsigned int j = 0;j < ImageDimension;++j)
487  resDiff(i,j) -= tmpPos[j];
488  }
489 
490  deltaMatrix = vnl_matrix_inverse <double> (deltaMatrix);
491 
492  for (unsigned int i = 0;i < ImageDimension;++i)
493  {
494  for (unsigned int j = 0;j < ImageDimension;++j)
495  {
496  jacMatrix(j,i) = 0;
497  for (unsigned int k = 0;k < ImageDimension;++k)
498  jacMatrix(j,i) += deltaMatrix(i,k)*resDiff(k,j);
499  }
500  }
501 
502  if (startIndDef[2] == (endIndDef[2]-1))
503  jacMatrix(2,2) = 1;
504 
505  double jacValue = vnl_det(jacMatrix);
506  if (jacValue < 0)
507  jacValue = 0;
508 
509  return jacValue;
510 }
