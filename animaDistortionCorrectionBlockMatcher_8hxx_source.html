189  correctionDirection[i] = this->GetReferenceImage()->GetDirection()(i,m_TransformDirection);
190

192  meanStructureTensor.fill(0);
193 
194  while (!fixedItr.IsAtEnd())
195  {
196  currentIndex = fixedItr.GetIndex();
197  std::fill(localGradient.begin(),localGradient.end(),0);
198 
199  for (unsigned int i = 0;i < InputImageType::ImageDimension;++i)
200  {
201  modifiedIndex = currentIndex;
202  modifiedIndex[i] = std::max(0,(int)(currentIndex[i] - 1));
203  double previousValue = this->GetReferenceImage()->GetPixel(modifiedIndex);
204  modifiedIndex[i] = std::min((int)(imageSize[i] - 1),(int)(currentIndex[i] + 1));
205  double postValue = this->GetReferenceImage()->GetPixel(modifiedIndex);
206 
207  for (unsigned int j = 0;j < InputImageType::ImageDimension;++j)
208  localGradient[j] += (postValue - previousValue) * orientationMatrix(j,i) / (2.0 * imageSpacing[i]);

212  for (unsigned int j = i;j < InputImageType::ImageDimension;++j)
213  {

215  if (j != i)
216  meanStructureTensor(j,i) = meanStructureTensor(i,j);
217  }
218 
219  ++fixedItr;
220  }
221
